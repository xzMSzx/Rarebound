The loader must automatically collect all pages.

SECTION 1 — REMOVE SINGLE PAGE FETCH

Find the current API request:

const url = https://api.pokemontcg.io/v2/cards?q=set.id:${setId}&pageSize=250

Remove the single fetch logic.

---

SECTION 2 — CREATE PAGINATED FETCH LOOP

Replace with:

async function fetchSetCards(setId) {

let page = 1
let allCards = []

while (true) {

const url =
https://api.pokemontcg.io/v2/cards?q=set.id:${setId}&pageSize=250&page=${page}

console.log("Fetching page:", page)

const response = await fetch(url)

if (!response.ok) {
throw new Error("API request failed")
}

const json = await response.json()

if (!Array.isArray(json.data)) {
throw new Error("Malformed API response")
}

if (json.data.length === 0) {
break
}

allCards = allCards.concat(json.data)

if (json.data.length < 250) {
break
}

page++

}

console.log("Total cards fetched:", allCards.length)

return allCards

}

---

SECTION 3 — IMAGE NORMALIZATION

After fetching all pages add:

const cards = allCards.filter(card =>
card.images && (card.images.small || card.images.large)
)

cards.forEach(card => {

if (!card.images.small && card.images.large) {
card.images.small = card.images.large
}

})

---

SECTION 4 — FINAL RETURN

Ensure the function ends with:

console.log("Loader returning cards:", cards.length)

return cards
