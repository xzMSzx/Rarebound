STEP 1 — CREATE PLAYER STATE MODULE

Create file:

/state/playerState.js

Paste EXACTLY:

let playerState = {
balance: 120.00,
collection: {}
}

export function loadPlayerState(){
const saved = localStorage.getItem("playerState")
if(saved){
playerState = JSON.parse(saved)
}
}

export function savePlayerState(){
localStorage.setItem("playerState", JSON.stringify(playerState))
}

export function getBalance(){
return playerState.balance
}

export function spendBalance(amount){
if(playerState.balance < amount){
return false
}
playerState.balance -= amount
savePlayerState()
return true
}

export function addCard(card){
if(!playerState.collection[card.id]){
playerState.collection[card.id] = {
count: 1,
rarity: card.rarityType,
setId: card.set.id,
name: card.name
}
} else {
playerState.collection[card.id].count++
}
savePlayerState()
}

export function getCollection(){
return playerState.collection
}
